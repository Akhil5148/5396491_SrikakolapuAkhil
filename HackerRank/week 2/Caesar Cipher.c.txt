#include <stdio.h>
#include <string.h>
#include <ctype.h>

char* caesarCipher(char s[], int k) {
    int n = strlen(s);
    k = k % 26; // Normalize rotation factor
    
    for (int i = 0; i < n; i++) {
        if (isalpha(s[i])) { // Only encrypt letters
            if (isupper(s[i])) {
                s[i] = 'A' + (s[i] - 'A' + k) % 26;
            } else {
                s[i] = 'a' + (s[i] - 'a' + k) % 26;
            }
        }
        // Non-alphabetic characters remain unchanged
    }
    return s;
}

int main() {
    int n, k;
    char s[1000];

    scanf("%d", &n);
    scanf("%s", s);
    scanf("%d", &k);

    printf("%s\n", caesarCipher(s, k));
    return 0;
}
