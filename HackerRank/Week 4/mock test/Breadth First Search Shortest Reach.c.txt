#include <stdio.h>
#include <stdlib.h>

#define MAXN 1001
#define QUEUE_SIZE 10000

int adj[MAXN][MAXN], deg[MAXN];
int dist[MAXN], q[QUEUE_SIZE], vis[MAXN];

int main() {
    int Q;
    scanf("%d", &Q);
    while(Q--) {
        int n, m;
        scanf("%d %d", &n, &m);
        for(int i = 1; i <= n; i++) deg[i] = 0;
        for(int i = 0; i < m; i++) {
            int u, v;
            scanf("%d %d", &u, &v);
            int ok = 1;
            for(int j=0;j<deg[u];j++)
                if(adj[u][j]==v) ok=0;
            if(ok){
                adj[u][deg[u]++] = v;
                adj[v][deg[v]++] = u;
            }
        }
        int s;
        scanf("%d", &s);
        for(int i = 1; i <= n; i++) dist[i] = -1, vis[i]=0;
        int front = 0, back = 0;
        q[back++] = s;
        vis[s] = 1;
        dist[s] = 0;
        while(front < back) {
            int u = q[front++];
            for(int i = 0; i < deg[u]; i++) {
                int v = adj[u][i];
                if(!vis[v]) {
                    vis[v] = 1;
                    dist[v] = dist[u] + 6;
                    q[back++] = v;
                }
            }
        }
        int first = 1;
        for(int i = 1; i <= n; i++) {
            if(i == s) continue;
            if(!first) printf(" ");
            printf("%d", dist[i]);
            first = 0;
        }
        printf("\n");
    }
    return 0;
}
